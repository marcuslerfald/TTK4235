219 static void state_moving(fsm_event_t event)
220 {
221  int requested_floor;
222  direction_t requested_direction;
223  queue_get_next(m_fsm.current_floor, m_fsm.direction, &requested_floor, &requested_direction);
224 
225  switch(event)
226  {
227  case EVENT_ENTRY:
228  m_fsm.state_name = STATE_MOVING;
229 
230  // Determine direction to move
231  if(requested_floor > m_fsm.current_floor)
232  {
233  elevator_go_up();
234 
235  if(m_fsm.position == POSITION_ON)
236  {
237  m_fsm.position = POSITION_ABOVE;
238  }
239  }
240  else if(requested_floor < m_fsm.current_floor)
241  {
242  elevator_go_down();
243 
244  if(m_fsm.position == POSITION_ON)
245  {
246  m_fsm.position = POSITION_BELOW;
247  }
248  }
249  else
250  {
251  if(elevator_on_floor())
252  {
253  // Elevator on correct floor
254 
255  // Do nothing, FSM will receive EVENT_FLOOR_X shortly, then open door
256  }
257  else
258  {
259  // Elevator is somewhere inbetween floors,
260  // due to previous emergency stop
261 
262  if(m_fsm.position == POSITION_ABOVE)
263  {
264  elevator_go_down();
265  }
266  else
267  {
268  elevator_go_up();
269  }
270  }
271  }
272  break;
273 
274  case EVENT_STOP_BUTTON_PRESSED:
275  fsm_transition(&state_emergency_stop);
276  break;
277 
278  // Implicit falltrough intentional,
279  // handled the same way
280  case EVENT_FLOOR_1:
281  case EVENT_FLOOR_2:
282  case EVENT_FLOOR_3:
283  case EVENT_FLOOR_4:
284  m_fsm.current_floor = (int)event;
285 
286  if(m_fsm.current_floor == requested_floor)
287  {
288  // Clear order lights and queue request
289  if(requested_direction == DIRECTION_UP)
290  {
291  hardware_command_order_light(requested_floor, HARDWARE_ORDER_UP, false);
292  queue_remove_request(m_fsm.current_floor, DIRECTION_UP);
293  }
294  else
295  {
296  hardware_command_order_light(requested_floor, HARDWARE_ORDER_DOWN, false);
297  queue_remove_request(m_fsm.current_floor, DIRECTION_DOWN);
298  }
299 
300  // Always clear internal order
301  hardware_command_order_light(requested_floor, HARDWARE_ORDER_INSIDE, false);
302 
303  m_fsm.position = POSITION_ON;
304 
305  fsm_transition(&state_door_open);
306  }
307  else
308  {
309  if(m_fsm.direction == DIRECTION_UP)
310  {
311  m_fsm.position = POSITION_ABOVE;
312  }
313  else
314  {
315  m_fsm.position = POSITION_BELOW;
316  }
317  }
318 
319  break;
320 
321  case EVENT_EXIT:
322  hardware_command_movement(HARDWARE_MOVEMENT_STOP);
323  break;
324  default:
325  break;
326  }
